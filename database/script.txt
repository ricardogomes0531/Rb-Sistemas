use rb_sistemas;
create table estabelecimento(id int primary key auto_increment, nome varchar(200) not null, documento varchar(20) not null, tipo_documento char(4) not null, categoria int not null);
create table fornecedor(id int primary key auto_increment, nome varchar(300) not null, descricao varchar(300), estabelecimentoId int not null, dataInclusao datetime, usuarioInclusao varchar(300), dataAlteracao datetime, usuarioAlteracao varchar(300));
