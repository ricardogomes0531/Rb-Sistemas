use rb_sistemas;
create table estabelecimento(id int primary key auto_increment, nome varchar(200) not null, documento varchar(20) not null, tipo_documento char(4) not null, categoria int not null);
create table fornecedor(id int primary key auto_increment, idEstabelecimento int not null, nome varchar(300) not null, descricao varchar(500), email varchar(300) not null, ddd_celular char(3), celular varchar(15), ddd_telefone char(3), telefone varchar(15), uf char(2), formaPagamentoPrincipal varchar(50), dataInsersao datetime, usuarioInsersao varchar(300), dataAlteracao datetime, usuarioAlteracao varchar(300));

create table Cliente(CodigoCliente int auto_increment primary key, Nome varchar(200), Email varchar(200), DddCelular char(2), Celular char(9), DataNascimento date);

CREATE TABLE Logs (Id INT AUTO_INCREMENT PRIMARY KEY, Timestamp DATETIME, Level VARCHAR(50), Message TEXT, Exception TEXT, Properties TEXT);
create table Empresa(CodigoEmpresa int auto_increment primary key, Nome varchar(200) not null, RazaoSocial varchar(500) not null, Documento varchar(20), DataInsersao datetime, UsuarioInsersao varchar(300), DataAlteracao datetime, UsuarioAlteracao varchar(300));
create table Usuarios(CodigoUsuario int auto_increment primary key, Nome varchar(200) not null, Password varchar(500) not null, Perfil varchar(30) not null, CodigoEmpresa int not null, DataInsersao datetime, UsuarioInsersao varchar(300), DataAlteracao datetime, UsuarioAlteracao varchar(300));
